//stock market analyzer
//write a function market_analyze that takes in an array of prices for a certain stock from some time in the past and returns the maximum profit you could have made by buying 1 share of that stock and selling it at some later time
//if your function market_analyze([10,20,90,15,7,89,2]), the function would return 82 which would be the profit if you bought the stock when the price was 7 and sold it when it was 89.
var biggestDiff = 0
var max = 0;
var min = 0;
function market_analyze(array) {
for (var i = 0; i < array.length; i++) {
biggestDiff = array.prototype.filter(Math.max(max,array[i]) - (Math.min(min,array[i])));
}
return biggestDiff;
}
market_analyze([10,20,90,15,7,89,2]);

str = str.toLowerCase();
 var vowels = "aeiouy";
 var cnt = 0;
 for (var i = 0; i < str.length; i++) {
   if (vowels.indexOf(str[i]) == "y") {
 	cnt = cnt - 0.25;
 } if (vowels.indexOf(str[i]) > -1) 
      cnt++;
 } 
 return cnt;             
}
